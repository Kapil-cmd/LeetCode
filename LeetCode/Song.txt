    static List<(double timestamp, string text)> LoadLyrics(string path)
    {
        var list = new List<(double timestamp, string text)>();
        var lines = File.ReadAllLines(path);

        foreach (var line in lines)
        {
            var match = Regex.Match(line, @"\[(\d{2}):(\d{2})\.(\d{2})\](.*)");
            if (match.Success)
            {
                int min = int.Parse(match.Groups[1].Value);
                int sec = int.Parse(match.Groups[2].Value);
                int ms = int.Parse(match.Groups[3].Value);
                string text = match.Groups[4].Value.Trim();

                double timestamp = min * 60 + sec + ms / 100.0;
                list.Add((timestamp, text));
            }
        }

        return list;
    }
